\item turning \texorpdfstring{$<$}{<} 0\+: ownship is turning left, turning \texorpdfstring{$>$}{>} 0\+: ownship is turning right, turning = 0\+: do not make any turning assumption about the ownship.
\item accelerating \texorpdfstring{$<$}{<} 0\+: ownship is decelerating, accelerating \texorpdfstring{$>$}{>} 0\+: ownship is accelerating, accelerating = 0\+: do not make any accelerating assumption about the ownship.
\item climbing \texorpdfstring{$<$}{<} 0\+: ownship is descending, climbing \texorpdfstring{$>$}{>} 0\+: ownship is climbing, climbing = 0\+: do not make any climbing assumption about the ownship. NOTES\+:
\end{DoxyItemize}
