963template <typename T>T parameter_data(const ParameterData& p, const std::string& key, Function<const std::string&,T> &f){
964 T default_val = f.apply(key);
965 if (p.contains(key)) {
966 return default_val;
967 }
968 DaidalusParameters::aliasestype::const_iterator find_ptr = DaidalusParameters::getAliases().find(key);
969 if (find_ptr == DaidalusParameters::getAliases().end()) {
970 return default_val;
971 }
972 std::vector<std::string>::const_iterator alias_ptr;
973 for (alias_ptr = find_ptr->second.begin(); alias_ptr != find_ptr->second.end(); ++alias_ptr) {
974 if (p.contains(*alias_ptr)) {
975 return f.apply(*alias_ptr);
976 }
977 }
978 return default_val;
979}
