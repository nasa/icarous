133 /*
134 * Set Daidalus object such that
135 * - Configure two alerters (Phase I and Phase II) as defined as in RTCA DO-365A
136 * - Maneuver guidance logic assumes kinematic maneuvers
137 * - Turn rate is set to 3 deg/s, when type is true, and to 1.5 deg/s
138 * when type is false.
139 * - Configure Sensor Uncertainty Migitation (SUM) when sum is true
140 * - Bands don't saturate until NMAC
141 */
142 void set_DO_365A(bool type=true, bool sum=true);
143
144 /*
145 * Set Daidalus object such that
146 * - Configure two alerters (Phase I, Phase II, and Non-Cooperative) as defined as in RTCA DO-365B
147 * - Maneuver guidance logic assumes kinematic maneuvers
148 * - Turn rate is set to 3 deg/s, when type is true, and to 1.5 deg/s
149 * when type is false.
150 * - Configure Sensor Uncertainty Migitation (SUM) when sum is true
151 * - Bands don't saturate until NMAC
152 */
153 void set_DO_365B(bool type=true, bool sum=true);
154
155 /*
156 * Set DAIDALUS object such that
157 * - Alerting thresholds are buffered
158 * - Maneuver guidance logic assumes kinematic maneuvers
159 * - Turn rate is set to 3 deg/s, when type is true, and to 1.5 deg/s
160 * when type is false.
161 * - Bands don't saturate until NMAC
162 */
