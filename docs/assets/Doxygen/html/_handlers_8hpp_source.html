316class TrafficConflictHandler: public EventHandler<CognitionState_t>{
317 public:
318 double startTime;
319
320 retVal_e Initialize(CognitionState_t* state){
321 LogMessage(state,"[HANDLER] | Traffic Conflict");
323 state->resolutionStartSpeed = state->velocity.gs();
324 state->prevResSpeed = state->resolutionStartSpeed;
325 state->prevResTrack = state->hdg;
326 state->prevResAlt = larcfm::Units::to(larcfm::Units::m,state->position.alt());
327 state->prevResVspeed = larcfm::Units::to(larcfm::Units::mps,state->velocity.vs());
328 state->preferredSpeed = state->prevResSpeed;
329 state->preferredTrack = state->prevResTrack;
330 state->preferredAlt = state->prevResAlt;
331 state->trafficConflictStartTime = state->utcTime;
332
334 state->resType = GetResolutionType(state);
335
337 int ind = state->resType;
338 if (state->validResolution[ind]){
339
340 std::string message = " against ";
341 for(auto id: state->conflictTraffics){
342 if (id != ""){
343 message += id + ", ";
344 }
345 }
346
347
348 if (state->resType == SPEED_RESOLUTION) {
349 LogMessage(state, "[STATUS] | "+ eventName +" | Resolving traffic conflict with speed resolution" + message);
350 }
351 else if (state->resType == ALTITUDE_RESOLUTION) {
352 LogMessage(state, "[STATUS] | "+ eventName +" | Resolving traffic conflict with altitude resolution" + message);
353 }
354 else if (state->resType == TRACK_RESOLUTION) {
355 LogMessage(state, "[STATUS] | "+ eventName +" | Resolving traffic conflict with track resolution" + message);
356 LogMessage(state, "[MODE] | Guidance Vector Request");
357 }
358 else if(state->resType == VERTICALSPEED_RESOLUTION ){
359 LogMessage(state, "[STATUS] | "+ eventName +" | Resolving traffic conflict with vertical speed resolution" + message);
360 LogMessage(state, "[MODE] | Guidance Vector Request");
361 }
362 }else{
364 return SHUTDOWN;
365 }
366
368 if(!state->parameters.active){
369 return SHUTDOWN;
370 }
371
372 return SUCCESS;
373 }
374
375 retVal_e Execute(CognitionState_t *state) {
376
377 int resolutionType = state->resType;
378
380 if (state->recovery[TRACK_RESOLUTION] &&
381 state->recovery[SPEED_RESOLUTION] &&
382 state->recovery[VERTICALSPEED_RESOLUTION]) {
383 if (resolutionType == TRACK_SPEED_VS_RESOLUTION) {
384 resolutionType = TRACK_RESOLUTION;
385 }
386 }
387
389 if(state->missionPlan == "Plan0"){
390 state->nextWpId[state->missionPlan] = state->nextFeasibleWpId;
391 }
392 int index = state->nextWpId[state->missionPlan];
393 larcfm::Position target = GetPlan(&state->flightPlans,state->missionPlan)->getPos(index);
394
399 switch (resolutionType) {
400 case SPEED_RESOLUTION: {
401 if (fabs(state->preferredSpeed - state->prevResSpeed) >= 0.1) {
403 SetGuidanceSpeedCmd(state,state->activePlan->getID(), state->preferredSpeed);
404 state->prevResSpeed = state->preferredSpeed;
405 }
406 state->returnSafe = ComputeTargetFeasibility(state,target);
407 break;
408 }
409
410 case ALTITUDE_RESOLUTION:
411 {
412 double climb_rate = 0.0;
413 double diff;
414
415 double current_alt = state->position.alt();
416 double alt_pref = state->preferredAlt;
417 if(state->todAltitude > 0){
418 alt_pref = state->todAltitude;
419 }
420 bool newTargetAlt = fabs(alt_pref - state->prevResAlt) > 1e-3;
421 bool prevTargetReached = fabs(current_alt - state->prevResAlt) < 10;
427 if ((newTargetAlt && prevTargetReached) || state->newAltConflict)
428 {
429 state->newAltConflict = false;
431 SetGuidanceAltCmd(state,state->activePlan->getID(), alt_pref, 1);
432 state->prevResAlt = alt_pref;
433 }
434
435 state->returnSafe = ComputeTargetFeasibility(state,target);
436
437 break;
438 }
439
440 case TRACK_RESOLUTION:
441 {
442 double speed = state->resolutionStartSpeed;
443 double climb_rate = 0;
448 SetGuidanceVelCmd(state,state->preferredTrack, speed, climb_rate);
449 state->prevResTrack = state->preferredTrack;
450 state->returnSafe = ComputeTargetFeasibility(state,target);
452 if(!state->parameters.return2NextWP){
453 state->returnSafe &= ComputeTargetFeasibility(state,state->clstPoint);
454 state->closestPointFeasible = state->returnSafe;
455 }
456 break;
457 }
458
459 case VERTICALSPEED_RESOLUTION:
460 {
461 double speed = state->resolutionStartSpeed;
462 double res_up = state->resVUp;
463 double res_down = state->resVDown;
464 // If there is a valid up resolution, execute up resolution.
465 // else execute the down resolution. If 0 vertical speed is possible,
466 // that is preferred over the up or down resolutions.
467 if (!std::isinf(res_up) && !std::isnan(res_up))
468 {
469 if (res_up >= 1e-3)
470 {
471 SetGuidanceVelCmd(state,state->hdg, speed, -res_up);
472 state->prevResVspeed = res_up;
473 }
474 else
475 {
476 SetGuidanceVelCmd(state,state->hdg, speed, 0.0);
477 state->prevResVspeed = 0.0;
478 }
479 }
480 else if (!std::isinf(res_down) && !std::isnan(res_down))
481 {
482 if (res_down <= -1e-3)
483 {
484 SetGuidanceVelCmd(state,state->hdg, speed, -res_down);
485 state->prevResVspeed = res_down;
486 }
487 else
488 {
489 SetGuidanceVelCmd(state,state->hdg, speed, 0.0);
490 state->prevResVspeed = 0.0;
491 }
492 }
493 else
494 {
495 SetGuidanceVelCmd(state,state->hdg, speed, state->prevResVspeed);
496 }
497
498 uint8_t val;
499 if (state->vsBandsNum > 0)
500 {
501 val = 0;
502 }
503 else
504 {
505 val = 1;
506 }
507
508 state->returnSafe = (bool)val;
509 break;
510 }
512 case TRACK_SPEED_VS_RESOLUTION:
513 {
514 double outTrack = state->preferredTrack;
515 double outSpeed = state->preferredSpeed;
516 double outVS = state->preferredVSpeed;
517 if (state->recovery[TRACK_RESOLUTION])
518 {
519 outTrack = std::fmod(360 + state->velocity.trk() * 180 / M_PI, 360);
520 }
521
522 if (state->recovery[SPEED_RESOLUTION])
523 {
524 outSpeed = state->velocity.gs();
525 }
526
527 if (state->recovery[VERTICALSPEED_RESOLUTION])
528 {
529 outVS = state->velocity.vs();
530 }
531
532 SetGuidanceVelCmd(state,outTrack, outSpeed, outVS);
533 state->returnSafe = ComputeTargetFeasibility(state,target);
534 }
535
536 default:
537 {
538
539 break;
540 }
541 }
542
544 bool conflict = state->trafficConflict || (!state->returnSafe);
545 if(conflict){
546 return INPROGRESS;
547 }else{
548 startTime = state->utcTime;
549 return SUCCESS;
550 }
551 }
552
553 retVal_e Terminate(CognitionState_t* state){
554
556 if(state->utcTime - startTime < state->parameters.persistenceTime){
557 return INPROGRESS;
558 }
559 std::string planid = state->activePlan->getID();
560 if(state->resType == SPEED_RESOLUTION){
566 SetGuidanceSpeedCmd(state,state->activePlan->getID(),state->resolutionStartSpeed);
567 SetGuidanceFlightPlan(state,planid,state->nextWpId[planid]);
568 LogMessage(state,"[HANDLER] | Engage nominal plan");
569 }else if(state->resType == ALTITUDE_RESOLUTION){
571 double alt = state->activePlan->getPos(state->nextWpId[planid]).alt();
572 if(fabs(state->prevResAlt - alt) > 1e-3){
573 SetGuidanceAltCmd(state,planid,alt,1);
574 state->prevResAlt = alt;
575 SetGuidanceFlightPlan(state,planid,state->nextWpId[planid]);
576 LogMessage(state,"[HANDLER] | Engage nominal plan");
577 }
578 }else{
580 if(!state->parameters.returnVector){
581 ExecuteHandler(MAKE_HANDLER(ReturnToMission),"PostTrafficConflict");
582 }else{
583 ExecuteHandler(MAKE_HANDLER(Vector2Mission),"PostTrafficConflict",priority-0.6);
584 }
585 }
586 std::string message = " against ";
587 for(auto id: state->conflictTraffics){
588 if (id != ""){
589 message += id + ", ";
590 }
591 }
592 SendStatus(state,(char*)"IC:traffic conflict resolved",6);
593 LogMessage(state,"[RESOLVED] | Traffic conflict resolved"+message);
594 state->conflictTraffics.clear();
595 return SUCCESS;
596 }
