311 retVal_e Initialize(CognitionState_t* state){
313 state->resolutionStartSpeed = state->velocity.gs();

327 int ind = state->resType;
328 if (state->validResolution[ind]){

330 std::string message = " against ";

334 }

338 if (state->resType == SPEED_RESOLUTION) {

340 }
341 else if (state->resType == ALTITUDE_RESOLUTION) {

365 retVal_e Execute(CognitionState_t *state) {
366
367 int resolutionType = state->resType;
368
370 if (state->recovery[TRACK_RESOLUTION] &&
371 state->recovery[SPEED_RESOLUTION] &&
372 state->recovery[VERTICALSPEED_RESOLUTION]) {
373 if (resolutionType == TRACK_SPEED_VS_RESOLUTION) {
374 resolutionType = TRACK_RESOLUTION;
375 }
376 }
377
379 if(state->missionPlan == "Plan0"){
380 state->nextWpId[state->missionPlan] = state->nextFeasibleWpId;
381 }
382 int index = state->nextWpId[state->missionPlan];
383 larcfm::Position target = GetPlan(&state->flightPlans,state->missionPlan)->getPos(index);
384
389 switch (resolutionType) {
390 case SPEED_RESOLUTION: {
391 if (fabs(state->preferredSpeed - state->prevResSpeed) >= 0.1) {
393 SetGuidanceSpeedCmd(state,state->activePlan->getID(), state->preferredSpeed);
394 state->prevResSpeed = state->preferredSpeed;
395 }
396 state->returnSafe = ComputeTargetFeasibility(state,target);
397 break;
398 }
399
400 case ALTITUDE_RESOLUTION:
401 {
402 double climb_rate = 0.0;
403 double diff;
404
405 double current_alt = state->position.alt();
406 double alt_pref = state->preferredAlt;
407 if(state->todAltitude > 0){
408 alt_pref = state->todAltitude;
409 }
410 bool newTargetAlt = fabs(alt_pref - state->prevResAlt) > 1e-3;
411 bool prevTargetReached = fabs(current_alt - state->prevResAlt) < 10;
417 if ((newTargetAlt && prevTargetReached) || state->newAltConflict)
418 {
419 state->newAltConflict = false;
421 SetGuidanceAltCmd(state,state->activePlan->getID(), alt_pref, 1);
422 state->prevResAlt = alt_pref;
423 }
424
425 state->returnSafe = ComputeTargetFeasibility(state,target);
426
427 break;
428 }
429
430 case TRACK_RESOLUTION:
431 {
432 double speed = state->resolutionStartSpeed;
433 double climb_rate = 0;
438 SetGuidanceVelCmd(state,state->preferredTrack, speed, climb_rate);
439 state->prevResTrack = state->preferredTrack;
440 state->returnSafe = ComputeTargetFeasibility(state,target);
442 if(!state->parameters.return2NextWP){
443 state->returnSafe &= ComputeTargetFeasibility(state,state->clstPoint);
444 state->closestPointFeasible = state->returnSafe;
445 }
446 break;
447 }
448
449 case VERTICALSPEED_RESOLUTION:
450 {
451 double speed = state->resolutionStartSpeed;
452 double res_up = state->resVUp;
453 double res_down = state->resVDown;
454 // If there is a valid up resolution, execute up resolution.
455 // else execute the down resolution. If 0 vertical speed is possible,
456 // that is preferred over the up or down resolutions.
457 if (!std::isinf(res_up) && !std::isnan(res_up))
458 {
459 if (res_up >= 1e-3)
460 {
461 SetGuidanceVelCmd(state,state->hdg, speed, -res_up);
462 state->prevResVspeed = res_up;
463 }
464 else
465 {
466 SetGuidanceVelCmd(state,state->hdg, speed, 0.0);
467 state->prevResVspeed = 0.0;
468 }
469 }
470 else if (!std::isinf(res_down) && !std::isnan(res_down))
471 {
472 if (res_down <= -1e-3)
473 {
474 SetGuidanceVelCmd(state,state->hdg, speed, -res_down);
475 state->prevResVspeed = res_down;
476 }
477 else
478 {
479 SetGuidanceVelCmd(state,state->hdg, speed, 0.0);
480 state->prevResVspeed = 0.0;
481 }
482 }
483 else
484 {
485 SetGuidanceVelCmd(state,state->hdg, speed, state->prevResVspeed);
486 }
487
488 uint8_t val;
489 if (state->vsBandsNum > 0)
490 {
491 val = 0;
492 }
493 else
494 {
495 val = 1;
496 }
497
498 state->returnSafe = (bool)val;
499 break;
500 }
502 case TRACK_SPEED_VS_RESOLUTION:
503 {
504 double outTrack = state->preferredTrack;
505 double outSpeed = state->preferredSpeed;
506 double outVS = state->preferredVSpeed;
507 if (state->recovery[TRACK_RESOLUTION])
508 {
509 outTrack = std::fmod(360 + state->velocity.trk() * 180 / M_PI, 360);
510 }
511
512 if (state->recovery[SPEED_RESOLUTION])
513 {
514 outSpeed = state->velocity.gs();
515 }
516
517 if (state->recovery[VERTICALSPEED_RESOLUTION])
518 {
519 outVS = state->velocity.vs();
520 }
521
522 SetGuidanceVelCmd(state,outTrack, outSpeed, outVS);
523 state->returnSafe = ComputeTargetFeasibility(state,target);
524 }
525
526 default:
527 {
528
529 break;
530 }
531 }
532
534 bool conflict = state->trafficConflict || (!state->returnSafe);
535 if(conflict){
536 return INPROGRESS;
537 }else{
538 startTime = state->utcTime;
539 return SUCCESS;
540 }
541 }

543 retVal_e Terminate(CognitionState_t* state){
544
546 if(state->utcTime - startTime < state->parameters.persistenceTime){
547 return INPROGRESS;
548 }
549 std::string planid = state->activePlan->getID();
550 if(state->resType == SPEED_RESOLUTION){
556 SetGuidanceSpeedCmd(state,state->activePlan->getID(),state->resolutionStartSpeed);
557 SetGuidanceFlightPlan(state,planid,state->nextWpId[planid]);
558 LogMessage(state,"[HANDLER] | Engage nominal plan");
559 }else if(state->resType == ALTITUDE_RESOLUTION){
561 double alt = state->activePlan->getPos(state->nextWpId[planid]).alt();
562 if(fabs(state->prevResAlt - alt) > 1e-3){
563 SetGuidanceAltCmd(state,planid,alt,1);
564 state->prevResAlt = alt;
565 SetGuidanceFlightPlan(state,planid,state->nextWpId[planid]);
566 LogMessage(state,"[HANDLER] | Engage nominal plan");
567 }
568 }else{
570 if(!state->parameters.returnVector){
571 ExecuteHandler(MAKE_HANDLER(ReturnToMission),"PostTrafficConflict");
572 }else{
573 ExecuteHandler(MAKE_HANDLER(Vector2Mission),"PostTrafficConflict",priority-0.6);
574 }
575 }
576 std::string message = " against ";
577 for(auto id: state->conflictTraffics){
578 if (id != ""){
579 message += id + ", ";
580 }
581 }
582 SendStatus(state,(char*)"IC:traffic conflict resolved",6);
583 LogMessage(state,"[RESOLVED] | Traffic conflict resolved"+message);
584 state->conflictTraffics.clear();
585 return SUCCESS;
586 }
