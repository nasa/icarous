64 MAVLINK_HELPER uint8_t mavlink_sign_packet(mavlink_signing_t *signing,
65  uint8_t signature[MAVLINK_SIGNATURE_BLOCK_LEN],
66  const uint8_t *header, uint8_t header_len,
67  const uint8_t *packet, uint8_t packet_len,
68  const uint8_t crc[2])
69 {
70  mavlink_sha256_ctx ctx;
71  union {
72  uint64_t t64;
73  uint8_t t8[8];
74  } tstamp;
75  if (signing == NULL || !(signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING)) {
76  return 0;
77  }
78  signature[0] = signing->link_id;
79  tstamp.t64 = signing->timestamp;
80  memcpy(&signature[1], tstamp.t8, 6);
81  signing->timestamp++;
82 
83  mavlink_sha256_init(&ctx);
84  mavlink_sha256_update(&ctx, signing->secret_key, sizeof(signing->secret_key));
85  mavlink_sha256_update(&ctx, header, header_len);
86  mavlink_sha256_update(&ctx, packet, packet_len);
87  mavlink_sha256_update(&ctx, crc, 2);
88  mavlink_sha256_update(&ctx, signature, 7);
89  mavlink_sha256_final_48(&ctx, &signature[7]);
90 
91  return MAVLINK_SIGNATURE_BLOCK_LEN;
92 }

110 MAVLINK_HELPER bool mavlink_signature_check(mavlink_signing_t *signing,
111  mavlink_signing_streams_t *signing_streams,
112  const mavlink_message_t *msg)
113 {
114  if (signing == NULL) {
115  return true;
116  }
117  const uint8_t *p = (const uint8_t *)&msg->magic;
118  const uint8_t *psig = msg->signature;
119  const uint8_t *incoming_signature = psig+7;
120  mavlink_sha256_ctx ctx;
121  uint8_t signature[6];
122  uint16_t i;
123 
124  mavlink_sha256_init(&ctx);
125  mavlink_sha256_update(&ctx, signing->secret_key, sizeof(signing->secret_key));
126  mavlink_sha256_update(&ctx, p, MAVLINK_CORE_HEADER_LEN+1+msg->len);
127  mavlink_sha256_update(&ctx, msg->ck, 2);
128  mavlink_sha256_update(&ctx, psig, 1+6);
129  mavlink_sha256_final_48(&ctx, signature);
130  if (memcmp(signature, incoming_signature, 6) != 0) {
131  return false;
132  }
133 
134  // now check timestamp
135  union tstamp {
136  uint64_t t64;
137  uint8_t t8[8];
138  } tstamp;
139  uint8_t link_id = psig[0];
140  tstamp.t64 = 0;
141  memcpy(tstamp.t8, psig+1, 6);
142 
143  if (signing_streams == NULL) {
144  return false;
145  }
146 
147  // find stream
148  for (i=0; i<signing_streams->num_signing_streams; i++) {
149  if (msg->sysid == signing_streams->stream[i].sysid &&
150  msg->compid == signing_streams->stream[i].compid &&
151  link_id == signing_streams->stream[i].link_id) {
152  break;
153  }
154  }
155  if (i == signing_streams->num_signing_streams) {
156  if (signing_streams->num_signing_streams >= MAVLINK_MAX_SIGNING_STREAMS) {
157  // over max number of streams
158  return false;
159  }
160  // new stream. Only accept if timestamp is not more than 1 minute old
161  if (tstamp.t64 + 6000*1000UL < signing->timestamp) {
162  return false;
163  }
164  // add new stream
165  signing_streams->stream[i].sysid = msg->sysid;
166  signing_streams->stream[i].compid = msg->compid;
167  signing_streams->stream[i].link_id = link_id;
168  signing_streams->num_signing_streams++;
169  } else {
170  union tstamp last_tstamp;
171  last_tstamp.t64 = 0;
172  memcpy(last_tstamp.t8, signing_streams->stream[i].timestamp_bytes, 6);
173  if (tstamp.t64 <= last_tstamp.t64) {
174  // repeating old timestamp
175  return false;
176  }
177  }
178 
179  // remember last timestamp
180  memcpy(signing_streams->stream[i].timestamp_bytes, psig+1, 6);
181 
182  // our next timestamp must be at least this timestamp
183  if (tstamp.t64 > signing->timestamp) {
184  signing->timestamp = tstamp.t64;
185  }
186  return true;
187 }

202 MAVLINK_HELPER uint16_t mavlink_finalize_message_buffer(mavlink_message_t* msg, uint8_t system_id, uint8_t component_id,
203  mavlink_status_t* status, uint8_t min_length, uint8_t length, uint8_t crc_extra)
204 {
205  bool mavlink1 = (status->flags & MAVLINK_STATUS_FLAG_OUT_MAVLINK1) != 0;
206  bool signing = (!mavlink1) && status->signing && (status->signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING);
207  uint8_t signature_len = signing? MAVLINK_SIGNATURE_BLOCK_LEN : 0;
208  uint8_t header_len = MAVLINK_CORE_HEADER_LEN+1;
209  uint8_t buf[MAVLINK_CORE_HEADER_LEN+1];
210  if (mavlink1) {
211  msg->magic = MAVLINK_STX_MAVLINK1;
212  header_len = MAVLINK_CORE_HEADER_MAVLINK1_LEN+1;
213  } else {
214  msg->magic = MAVLINK_STX;
215  }
216  msg->len = mavlink1?min_length:_mav_trim_payload(_MAV_PAYLOAD(msg), length);
217  msg->sysid = system_id;
218  msg->compid = component_id;
219  msg->incompat_flags = 0;
220  if (signing) {
221  msg->incompat_flags |= MAVLINK_IFLAG_SIGNED;
222  }
223  msg->compat_flags = 0;
224  msg->seq = status->current_tx_seq;
225  status->current_tx_seq = status->current_tx_seq + 1;
226 
227  // form the header as a byte array for the crc
228  buf[0] = msg->magic;
229  buf[1] = msg->len;
230  if (mavlink1) {
231  buf[2] = msg->seq;
232  buf[3] = msg->sysid;
233  buf[4] = msg->compid;
234  buf[5] = msg->msgid & 0xFF;
235  } else {
236  buf[2] = msg->incompat_flags;
237  buf[3] = msg->compat_flags;
238  buf[4] = msg->seq;
239  buf[5] = msg->sysid;
240  buf[6] = msg->compid;
241  buf[7] = msg->msgid & 0xFF;
242  buf[8] = (msg->msgid >> 8) & 0xFF;
243  buf[9] = (msg->msgid >> 16) & 0xFF;
244  }
245 
246  msg->checksum = crc_calculate(&buf[1], header_len-1);
247  crc_accumulate_buffer(&msg->checksum, _MAV_PAYLOAD(msg), msg->len);
248  crc_accumulate(crc_extra, &msg->checksum);
249  mavlink_ck_a(msg) = (uint8_t)(msg->checksum & 0xFF);
250  mavlink_ck_b(msg) = (uint8_t)(msg->checksum >> 8);
251 
252  if (signing) {
253  mavlink_sign_packet(status->signing,
254  msg->signature,
255  (const uint8_t *)buf, header_len,
256  (const uint8_t *)_MAV_PAYLOAD(msg), msg->len,
257  (const uint8_t *)_MAV_PAYLOAD(msg)+(uint16_t)msg->len);
258  }
259 
260  return msg->len + header_len + 2 + signature_len;
261 }

290 MAVLINK_HELPER void _mav_finalize_message_chan_send(mavlink_channel_t chan, uint32_t msgid,
291  const char *packet,
292  uint8_t min_length, uint8_t length, uint8_t crc_extra)
293 {
294  uint16_t checksum;
295  uint8_t buf[MAVLINK_NUM_HEADER_BYTES];
296  uint8_t ck[2];
297  mavlink_status_t *status = mavlink_get_channel_status(chan);
298  uint8_t header_len = MAVLINK_CORE_HEADER_LEN;
299  uint8_t signature_len = 0;
300  uint8_t signature[MAVLINK_SIGNATURE_BLOCK_LEN];
301  bool mavlink1 = (status->flags & MAVLINK_STATUS_FLAG_OUT_MAVLINK1) != 0;
302  bool signing = (!mavlink1) && status->signing && (status->signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING);
303 
304  if (mavlink1) {
305  length = min_length;
306  if (msgid > 255) {
307  // can't send 16 bit messages
308  _mav_parse_error(status);
309  return;
310  }
311  header_len = MAVLINK_CORE_HEADER_MAVLINK1_LEN;
312  buf[0] = MAVLINK_STX_MAVLINK1;
313  buf[1] = length;
314  buf[2] = status->current_tx_seq;
315  buf[3] = mavlink_system.sysid;
316  buf[4] = mavlink_system.compid;
317  buf[5] = msgid & 0xFF;
318  } else {
319  uint8_t incompat_flags = 0;
320  if (signing) {
321  incompat_flags |= MAVLINK_IFLAG_SIGNED;
322  }
323  length = _mav_trim_payload(packet, length);
324  buf[0] = MAVLINK_STX;
325  buf[1] = length;
326  buf[2] = incompat_flags;
327  buf[3] = 0; // compat_flags
328  buf[4] = status->current_tx_seq;
329  buf[5] = mavlink_system.sysid;
330  buf[6] = mavlink_system.compid;
331  buf[7] = msgid & 0xFF;
332  buf[8] = (msgid >> 8) & 0xFF;
333  buf[9] = (msgid >> 16) & 0xFF;
334  }
335  status->current_tx_seq++;
336  checksum = crc_calculate((const uint8_t*)&buf[1], header_len);
337  crc_accumulate_buffer(&checksum, packet, length);
338  crc_accumulate(crc_extra, &checksum);
339  ck[0] = (uint8_t)(checksum & 0xFF);
340  ck[1] = (uint8_t)(checksum >> 8);
341 
342  if (signing) {
343  // possibly add a signature
344  signature_len = mavlink_sign_packet(status->signing, signature, buf, header_len+1,
345  (const uint8_t *)packet, length, ck);
346  }
347 
348  MAVLINK_START_UART_SEND(chan, header_len + 3 + (uint16_t)length + (uint16_t)signature_len);
349  _mavlink_send_uart(chan, (const char *)buf, header_len+1);
350  _mavlink_send_uart(chan, packet, length);
351  _mavlink_send_uart(chan, (const char *)ck, 2);
352  if (signature_len != 0) {
353  _mavlink_send_uart(chan, (const char *)signature, signature_len);
354  }
355  MAVLINK_END_UART_SEND(chan, header_len + 3 + (uint16_t)length + (uint16_t)signature_len);
356 }

480  return the crc_entry value for a msgid
481 */
482 #ifndef MAVLINK_GET_MSG_ENTRY
483 MAVLINK_HELPER const mavlink_msg_entry_t *mavlink_get_msg_entry(uint32_t msgid)
484 {
485  static const mavlink_msg_entry_t mavlink_message_crcs[] = MAVLINK_MESSAGE_CRCS;
486  /*
487  use a bisection search to find the right entry. A perfect hash may be better
488  Note that this assumes the table is sorted by msgid
489  */
490  uint32_t low=0, high=sizeof(mavlink_message_crcs)/sizeof(mavlink_message_crcs[0]);
491  while (low < high) {
492  uint32_t mid = (low+1+high)/2;
493  if (msgid < mavlink_message_crcs[mid].msgid) {
494  high = mid-1;
495  continue;
496  }
497  if (msgid > mavlink_message_crcs[mid].msgid) {
498  low = mid;
499  continue;
500  }
501  low = mid;
502  break;
503  }
504  if (mavlink_message_crcs[low].msgid != msgid) {
505  // msgid is not in the table
506  return NULL;
507  }
508  return &mavlink_message_crcs[low];
509 }
510 #endif // MAVLINK_GET_MSG_ENTRY

565 MAVLINK_HELPER uint8_t mavlink_frame_char_buffer(mavlink_message_t* rxmsg,
566  mavlink_status_t* status,
567  uint8_t c,
568  mavlink_message_t* r_message,
569  mavlink_status_t* r_mavlink_status)
570 {
571  /* Enable this option to check the length of each message.
572  This allows invalid messages to be caught much sooner. Use if the transmission
573  medium is prone to missing (or extra) characters (e.g. a radio that fades in
574  and out). Only use if the channel will only contain messages types listed in
575  the headers.
576  */
577 #ifdef MAVLINK_CHECK_MESSAGE_LENGTH
578 #ifndef MAVLINK_MESSAGE_LENGTH
579  static const uint8_t mavlink_message_lengths[256] = MAVLINK_MESSAGE_LENGTHS;
580 #define MAVLINK_MESSAGE_LENGTH(msgid) mavlink_message_lengths[msgid]
581 #endif
582 #endif
583 
584  int bufferIndex = 0;
585 
586  status->msg_received = MAVLINK_FRAMING_INCOMPLETE;
587 
588  switch (status->parse_state)
589  {
590  case MAVLINK_PARSE_STATE_UNINIT:
591  case MAVLINK_PARSE_STATE_IDLE:
592  if (c == MAVLINK_STX)
593  {
594  status->parse_state = MAVLINK_PARSE_STATE_GOT_STX;
595  rxmsg->len = 0;
596  rxmsg->magic = c;
597  status->flags &= ~MAVLINK_STATUS_FLAG_IN_MAVLINK1;
598  mavlink_start_checksum(rxmsg);
599  } else if (c == MAVLINK_STX_MAVLINK1)
600  {
601  status->parse_state = MAVLINK_PARSE_STATE_GOT_STX;
602  rxmsg->len = 0;
603  rxmsg->magic = c;
604  status->flags |= MAVLINK_STATUS_FLAG_IN_MAVLINK1;
605  mavlink_start_checksum(rxmsg);
606  }
607  break;
608 
609  case MAVLINK_PARSE_STATE_GOT_STX:
610  if (status->msg_received
611 /* Support shorter buffers than the
612  default maximum packet size */
613 #if (MAVLINK_MAX_PAYLOAD_LEN < 255)
614  || c > MAVLINK_MAX_PAYLOAD_LEN
615 #endif
616  )
617  {
618  status->buffer_overrun++;
619  _mav_parse_error(status);
620  status->msg_received = 0;
621  status->parse_state = MAVLINK_PARSE_STATE_IDLE;
622  }
623  else
624  {

633  } else {

681 #ifdef MAVLINK_CHECK_MESSAGE_LENGTH

691  }
692  break;

694  case MAVLINK_PARSE_STATE_GOT_MSGID1:

697  status->parse_state = MAVLINK_PARSE_STATE_GOT_MSGID2;
698  break;

700  case MAVLINK_PARSE_STATE_GOT_MSGID2:
701  rxmsg->msgid |= c<<16;
702  mavlink_update_checksum(rxmsg, c);
703  status->parse_state = MAVLINK_PARSE_STATE_GOT_MSGID3;
704 #ifdef MAVLINK_CHECK_MESSAGE_LENGTH
705  if (rxmsg->len != MAVLINK_MESSAGE_LENGTH(rxmsg->msgid))
706  {
707  _mav_parse_error(status);
708  status->parse_state = MAVLINK_PARSE_STATE_IDLE;
709  break;
710  }
711 #endif
712  break;
713 
714  case MAVLINK_PARSE_STATE_GOT_MSGID3:
715  _MAV_PAYLOAD_NON_CONST(rxmsg)[status->packet_idx++] = (char)c;
716  mavlink_update_checksum(rxmsg, c);
717  if (status->packet_idx == rxmsg->len)
718  {
719  status->parse_state = MAVLINK_PARSE_STATE_GOT_PAYLOAD;
720  }
721  break;
722

725  uint8_t crc_extra = e?e->crc_extra:0;
726  mavlink_update_checksum(rxmsg, crc_extra);
727  if (c != (rxmsg->checksum & 0xFF)) {
728  status->parse_state = MAVLINK_PARSE_STATE_GOT_BAD_CRC1;
729  } else {
730  status->parse_state = MAVLINK_PARSE_STATE_GOT_CRC1;
731  }

733

735  if (e && status->packet_idx < e->msg_len) {
736  memset(&_MAV_PAYLOAD_NON_CONST(rxmsg)[status->packet_idx], 0, e->msg_len - status->packet_idx);
737  }
738  break;
739  }
740 
741  case MAVLINK_PARSE_STATE_GOT_CRC1:
742  case MAVLINK_PARSE_STATE_GOT_BAD_CRC1:
743  if (status->parse_state == MAVLINK_PARSE_STATE_GOT_BAD_CRC1 || c != (rxmsg->checksum >> 8)) {
744  // got a bad CRC message
745  status->msg_received = MAVLINK_FRAMING_BAD_CRC;
746  } else {
747  // Successfully got message
748  status->msg_received = MAVLINK_FRAMING_OK;
749  }
750  rxmsg->ck[1] = c;
751 
752  if (rxmsg->incompat_flags & MAVLINK_IFLAG_SIGNED) {
753  status->parse_state = MAVLINK_PARSE_STATE_SIGNATURE_WAIT;
754  status->signature_wait = MAVLINK_SIGNATURE_BLOCK_LEN;
755 
756  // If the CRC is already wrong, don't overwrite msg_received,
757  // otherwise we can end up with garbage flagged as valid.
758  if (status->msg_received != MAVLINK_FRAMING_BAD_CRC) {
759  status->msg_received = MAVLINK_FRAMING_INCOMPLETE;
760  }
761  } else {
762  if (status->signing &&
763  (status->signing->accept_unsigned_callback == NULL ||
764  !status->signing->accept_unsigned_callback(status, rxmsg->msgid))) {
765 
766  // If the CRC is already wrong, don't overwrite msg_received.
767  if (status->msg_received != MAVLINK_FRAMING_BAD_CRC) {
768  status->msg_received = MAVLINK_FRAMING_BAD_SIGNATURE;
769  }
770  }
771  status->parse_state = MAVLINK_PARSE_STATE_IDLE;
772  memcpy(r_message, rxmsg, sizeof(mavlink_message_t));
773  }
774  break;
775  case MAVLINK_PARSE_STATE_SIGNATURE_WAIT:
776  rxmsg->signature[MAVLINK_SIGNATURE_BLOCK_LEN-status->signature_wait] = c;
777  status->signature_wait--;
778  if (status->signature_wait == 0) {
779  // we have the whole signature, check it is OK
780  bool sig_ok = mavlink_signature_check(status->signing, status->signing_streams, rxmsg);
781  if (!sig_ok &&
782  (status->signing->accept_unsigned_callback &&
783  status->signing->accept_unsigned_callback(status, rxmsg->msgid))) {
784  // accepted via application level override
785  sig_ok = true;
786  }
787  if (sig_ok) {
788  status->msg_received = MAVLINK_FRAMING_OK;
789  } else {
790  status->msg_received = MAVLINK_FRAMING_BAD_SIGNATURE;
791  }
792  status->parse_state = MAVLINK_PARSE_STATE_IDLE;
793  memcpy(r_message, rxmsg, sizeof(mavlink_message_t));
794  }
795  break;
796  }
797 
798  bufferIndex++;
799  // If a message has been sucessfully decoded, check index
800  if (status->msg_received == MAVLINK_FRAMING_OK)
801  {
802  //while(status->current_seq != rxmsg->seq)
803  //{
804  // status->packet_rx_drop_count++;
805  // status->current_seq++;
806  //}
807  status->current_rx_seq = rxmsg->seq;
808  // Initial condition: If no packet has been received so far, drop count is undefined
809  if (status->packet_rx_success_count == 0) status->packet_rx_drop_count = 0;
810  // Count this packet as received
811  status->packet_rx_success_count++;
812  }
813 
814  r_message->len = rxmsg->len; // Provide visibility on how far we are into current msg
815  r_mavlink_status->parse_state = status->parse_state;
816  r_mavlink_status->packet_idx = status->packet_idx;
817  r_mavlink_status->current_rx_seq = status->current_rx_seq+1;
818  r_mavlink_status->packet_rx_success_count = status->packet_rx_success_count;
819  r_mavlink_status->packet_rx_drop_count = status->parse_error;
820  r_mavlink_status->flags = status->flags;
821  status->parse_error = 0;
822 
823  if (status->msg_received == MAVLINK_FRAMING_BAD_CRC) {
824  /*
825  the CRC came out wrong. We now need to overwrite the
826  msg CRC with the one on the wire so that if the
827  caller decides to forward the message anyway that
828  mavlink_msg_to_send_buffer() won't overwrite the
829  checksum
830  */
831  r_message->checksum = rxmsg->ck[0] | (rxmsg->ck[1]<<8);
832  }
833 
834  return status->msg_received;
835 }

987 MAVLINK_HELPER uint8_t put_bitfield_n_by_index(int32_t b, uint8_t bits, uint8_t packet_index, uint8_t bit_index, uint8_t* r_bit_index, uint8_t* buffer)
988 {
989  uint16_t bits_remain = bits;
990  // Transform number into network order
991  int32_t v;
992  uint8_t i_bit_index, i_byte_index, curr_bits_n;
993 #if MAVLINK_NEED_BYTE_SWAP
994  union {
995  int32_t i;
996  uint8_t b[4];
997  } bin, bout;
998  bin.i = b;
999  bout.b[0] = bin.b[3];
1000  bout.b[1] = bin.b[2];
1001  bout.b[2] = bin.b[1];
1002  bout.b[3] = bin.b[0];
1003  v = bout.i;
1004 #else
1005  v = b;
1006 #endif
1007 
1008  // buffer in
1009  // 01100000 01000000 00000000 11110001
1010  // buffer out
1011  // 11110001 00000000 01000000 01100000
1012 
1013  // Existing partly filled byte (four free slots)
1014  // 0111xxxx
1015 
1016  // Mask n free bits
1017  // 00001111 = 2^0 + 2^1 + 2^2 + 2^3 = 2^n - 1
1018  // = ((uint32_t)(1 << n)) - 1; // = 2^n - 1
1019 
1020  // Shift n bits into the right position
1021  // out = in >> n;
1022 
1023  // Mask and shift bytes
1024  i_bit_index = bit_index;
1025  i_byte_index = packet_index;
1026  if (bit_index > 0)
1027  {
1028  // If bits were available at start, they were available
1029  // in the byte before the current index
1030  i_byte_index--;
1031  }
1032 
1033  // While bits have not been packed yet
1034  while (bits_remain > 0)
1035  {
1036  // Bits still have to be packed
1037  // there can be more than 8 bits, so
1038  // we might have to pack them into more than one byte
1039 
1040  // First pack everything we can into the current 'open' byte
1041  //curr_bits_n = bits_remain << 3; // Equals bits_remain mod 8
1042  //FIXME
1043  if (bits_remain <= (uint8_t)(8 - i_bit_index))
1044  {
1045  // Enough space
1046  curr_bits_n = (uint8_t)bits_remain;
1047  }
1048  else
1049  {
1050  curr_bits_n = (8 - i_bit_index);
1051  }
1052 
1053  // Pack these n bits into the current byte
1054  // Mask out whatever was at that position with ones (xxx11111)
1055  buffer[i_byte_index] &= (0xFF >> (8 - curr_bits_n));
1056  // Put content to this position, by masking out the non-used part
1057  buffer[i_byte_index] |= ((0x00 << curr_bits_n) & v);
1058 
1059  // Increment the bit index
1060  i_bit_index += curr_bits_n;
1061 
1062  // Now proceed to the next byte, if necessary
1063  bits_remain -= curr_bits_n;
1064  if (bits_remain > 0)
1065  {
1066  // Offer another 8 bits / one byte
1067  i_byte_index++;
1068  i_bit_index = 0;
1069  }
1070  }
1071 
1072  *r_bit_index = i_bit_index;
1073  // If a partly filled byte is present, mark this as consumed
1074  if (i_bit_index != 7) i_byte_index++;
1075  return i_byte_index - packet_index;
1076 }
